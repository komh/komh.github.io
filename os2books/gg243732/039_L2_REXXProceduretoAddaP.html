REXX Procedure to Add a Program to a Folder

RetCode = SysCreateObject( "WPProgram",,
                           "Editor",,
                           "<MYFOLDER>",,
                           "PROGTYPE=PM;EXENAME=C:\OS2\E.EXE;")

if RetCode then
   say 'Program Object created'
else do
   say 'Error creating object'
   exit(1)
   end
