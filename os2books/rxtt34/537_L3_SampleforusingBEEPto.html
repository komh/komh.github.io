                    /* number of repeats                              */
  repeatNo = 1
                    /* create a stem with note and duration           */
  J = 0
  do I = 1 to Words( Data ) by 2
    J = J + 1
    Note.J = SubWord( Data, I, 1 )
    Duration.J = SubWord( Data, I + 1, 1 )
  end I

  do a = 1 to repeatNo
    do play = 1 to J
      Call Beep Note.play, Duration.play
    end /* do play = 1 to J */
  end /* do a = 1 to repeatNo */
