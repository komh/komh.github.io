CC_SET_PARMS

This data structure contains information for the ccSendCommand function when using the CC_SET message. 

typedef struct _CC_SET_PARMS {
  ULONG     ulRows;
  ULONG     ulBackgroundColor;
  ULONG     ulTextColor;
  ULONG     ulXposition;
  ULONG     ulYposition;
  ULONG     ulColumns;
} CC_SET_PARMS;

typedef   CC _ SET _ PARMS   FAR   * PCC _ SET _ PARMS ;
