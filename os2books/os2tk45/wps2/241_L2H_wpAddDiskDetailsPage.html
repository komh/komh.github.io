wpAddDiskDetailsPage - Syntax

This instance method is called to allow the object to add the Details page to its Settings notebook. 

#define INCL_WINWORKPLACE
#include <os2.h>

WPDisk     *somSelf;       /*  Pointer to the object on which the method is being invoked. */
HWND        hwndNotebook;  /*  Settings notebook handle. */
ULONG       rc;            /*  Page identifier. */

rc = _wpAddDiskDetailsPage(somSelf, hwndNotebook);
