wpAllocMem - Syntax

This instance method is called to allocate memory for use by an object. 

#define INCL_WINWORKPLACE
#include <os2.h>

WPObject     *somSelf;   /*  Pointer to the object on which the method is being invoked. */
ULONG         cbBytes;   /*  Specifies the size, in bytes, of memory required. */
PULONG        prc;       /*  Pointer to error code. */
PBYTE         pbNewMem;  /*  Success indicator. */

pbNewMem = _wpAllocMem(somSelf, cbBytes, prc);
