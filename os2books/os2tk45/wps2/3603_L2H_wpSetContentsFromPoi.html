wpSetContentsFromPointer - Example Code

This example copies the contents of one data file to another. 

PVOID       pData;

pData = _wpQueryPointerFromContents(SourceDataFile);
if (pData)
{
   _wpSetContentsFromPointer(TargetDataFile,pData);
   DosFreeMem(pData);
}
