VISUALCONFIG

The Visual Configuration structure is required when creating an OpenGL context. 

typedef struct _VISUALCONFIG {
  ULONG                     vid;             /*  Visual ID. */
  BOOL                      rgba;
  int                       redSize;
  int                       greenSize;
  int                       blueSize;
  int                       alphaSize;
  ULONG                     redMask;
  ULONG                     greenMask;
  ULONG                     blueMask;
  ULONG                     accumRedSize;
  ULONG                     accumGreenSize;
  ULONG                     accumBlueSize;
  ULONG                     accumAlphaSize;
  BOOL                      doubleBuffer;
  BOOL                      stereo;
  int                       bufferSize;
  int                       depthSize;
  int                       stencilSize;
  int                       auxBuffers;
  int                       level;
  PVOID                     reserved;
  struct visualconfig      *next;
} VISUALCONFIG;

typedef   VISUALCONFIG   * PVISUALCONFIG ;
