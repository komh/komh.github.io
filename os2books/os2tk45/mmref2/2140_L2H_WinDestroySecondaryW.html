WinDestroySecondaryWindow - Syntax

This function eliminates a secondary window. 

#define INCL_SW
#include <os2.h>

HWND    hwnd;  /*  Secondary window to be eliminated. */
BOOL    rc;    /*  Return codes. */

rc = WinDestroySecondaryWindow(hwnd);
