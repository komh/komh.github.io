mmioCFClose - Syntax

This function closes a compound file that was opened by mmioCFOpen. 

#define INCL_MMIOOS2
#include <os2.h>

HMMCF    hmmcf;    /*  Compound-file handle. */
ULONG    ulFlags;  /*  Reserved. */
ULONG    rc;       /*  Return codes. */

rc = mmioCFClose(hmmcf, ulFlags);
