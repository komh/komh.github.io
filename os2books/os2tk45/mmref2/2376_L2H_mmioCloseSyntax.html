mmioClose - Syntax

This function closes a file that was opened by mmioOpen. 

#define INCL_MMIOOS2
#include <os2.h>

HMMIO     hmmio;    /*  Open file handle. */
USHORT    usFlags;  /*  Flags. */
USHORT    rc;       /*  Return codes. */

rc = mmioClose(hmmio, usFlags);
