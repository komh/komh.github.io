WinIsWPDServerReady - Syntax

This function returns the state of the Workplace Shell DSOM server. 

#define INCL_WPCLASS
#include <os2.h>

BOOL    fReady;  /*  DSOM server-state indicator. DSOM Server status. */

fReady = WinIsWPDServerReady();
