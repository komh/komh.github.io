WinDestroyObject - Syntax

This function is called to delete a workplace object. 

#define INCL_WINWORKPLACE
#include <os2.h>

HOBJECT    object;  /*  Handle to a workplace object. */
BOOL       rc;      /*  Success indicator. */

rc = WinDestroyObject(object);
