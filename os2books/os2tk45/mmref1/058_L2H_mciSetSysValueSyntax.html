mciSetSysValue - Syntax

This function sets the value of system-defined attributes. 

#include <os2.h>

USHORT    iSysValue;  /*  System attribute. */
PVOID     pValue;     /*  Pointer to value. */
BOOL      rc;         /*  Return code. */

rc = mciSetSysValue(iSysValue, pValue);
