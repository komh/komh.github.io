mciQuerySysValue - Syntax

This function queries the value of system-defined attributes. 

#define INCL_MCIOS2
#include <os2.h>

USHORT    iSysValue;  /*  System attribute. */
PVOID     pValue;     /*  Pointer to return field. */
BOOL      rc;         /*  Return code. */

rc = mciQuerySysValue(iSysValue, pValue);
