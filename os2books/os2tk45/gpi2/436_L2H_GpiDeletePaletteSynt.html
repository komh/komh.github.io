GpiDeletePalette - Syntax

This function deletes a color palette. 

#define INCL_GPILOGCOLORTABLE /* Or use INCL_GPI, INCL_PM, */
#include <os2.h>

HPAL    hpal;  /*  Palette handle. */
BOOL    rc;    /*  Success indicator. */

rc = GpiDeletePalette(hpal);
