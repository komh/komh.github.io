GpiSetSegmentAttrs - Syntax

This function sets a segment attribute. 

#define INCL_GPISEGMENTS /* Or use INCL_GPI, INCL_PM, */
#include <os2.h>

HPS     hps;         /*  Presentation-space handle. */
LONG    lSegid;      /*  Segment identifier. */
LONG    lAttribute;  /*  Segment attribute. */
LONG    lValue;      /*  Attribute value. */
BOOL    rc;          /*  Success indicator. */

rc = GpiSetSegmentAttrs(hps, lSegid, lAttribute,
       lValue);
