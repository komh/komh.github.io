GpiExcludeClipRectangle - Syntax

This function excludes a rectangle from the clipping region. 

#define INCL_GPIREGIONS /* Or use INCL_GPI, INCL_PM, */
#include <os2.h>

HPS       hps;            /*  Presentation-space handle. */
PRECTL    prclRectangle;  /*  Rectangle to be excluded. */
LONG      lComplexity;    /*  Complexity of clipping and error indicators. */

lComplexity = GpiExcludeClipRectangle(hps,
                prclRectangle);
