GpiSetCharSet - Example Code

This function sets the current value of the character-set attribute. 

#define INCL_GPIPRIMITIVES
#include <OS2.H>


HPS     hps;    /* Presentation space handle     */
LONG llcid = 32L;

GpiSetCharSet(hps, llcid);
