GpiTranslate - Syntax

This function applies a translation to a transform matrix. 

#define INCL_GPITRANSFORMS /* Or use INCL_GPI, INCL_PM, */
#include <os2.h>

HPS          hps;              /*  Presentation-space handle. */
PMATRIXLF    pmatlfArray;      /*  Transform matrix. */
LONG         lOptions;         /*  Transform options. */
PPOINTL      pptlTranslation;  /*  Translation. */
BOOL         rc;               /*  Success indicator. */

rc = GpiTranslate(hps, pmatlfArray, lOptions,
       pptlTranslation);
