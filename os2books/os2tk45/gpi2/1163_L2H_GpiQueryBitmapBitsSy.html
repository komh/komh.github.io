GpiQueryBitmapBits - Syntax

This function transfers data from a bit map to application storage. 

#define INCL_GPIBITMAPS /* Or use INCL_GPI, INCL_PM, */
#include <os2.h>

HPS             hps;             /*  Presentation-space handle. */
LONG            lScanStart;      /*  Starting line number. */
LONG            lScans;          /*  Number of scan lines to be returned. */
PBYTE           pbBuffer;        /*  Data area. */
PBITMAPINFO2    pbmiInfoTable;   /*  Bit-map information table. */
LONG            lScansReturned;  /*  Number of scan lines actually returned. */

lScansReturned = GpiQueryBitmapBits(hps, lScanStart,
                   lScans, pbBuffer, pbmiInfoTable);
