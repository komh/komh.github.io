GpiDeleteMetaFile - Syntax

This function deletes a metafile. 

#define INCL_GPIMETAFILES /* Or use INCL_GPI, INCL_PM, */
#include <os2.h>

HMF     hmf;  /*  Metafile handle. */
BOOL    rc;   /*  Success indicator. */

rc = GpiDeleteMetaFile(hmf);
