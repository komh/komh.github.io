GpiCreateLogColorTable - Syntax

This function defines the entries of the logical color table. 

#define INCL_GPILOGCOLORTABLE /* Or use INCL_GPI, INCL_PM, */
#include <os2.h>

HPS      hps;        /*  Presentation-space handle. */
ULONG    flOptions;  /*  Options. */
LONG     lFormat;    /*  Format of entries in the table. */
LONG     lStart;     /*  Starting index. */
LONG     lCount;     /*  Count of elements in alTable. */
PLONG    alTable;    /*  Start of the application data area. */
BOOL     rc;         /*  Success indicator. */

rc = GpiCreateLogColorTable(hps, flOptions,
       lFormat, lStart, lCount, alTable);
