GpiSetLineWidthGeom - Syntax

This function sets the current geometric line-width attribute. 

#define INCL_GPIPRIMITIVES /* Or use INCL_GPI, INCL_PM, */
#include <os2.h>

HPS     hps;         /*  Presentation-space handle. */
LONG    lLineWidth;  /*  Geometric line width. */
BOOL    rc;          /*  Success indicator. */

rc = GpiSetLineWidthGeom(hps, lLineWidth);
