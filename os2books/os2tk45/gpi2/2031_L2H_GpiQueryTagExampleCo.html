GpiQueryTag - Example Code

This function returns the current value of the tag identifier, as set by the GpiSetTag call. 

#define INCL_GPICORRELATION
#include <OS2.H>

HPS hps;         /*   Presentation-space */
                 /*   handle.            */

LONG lTag;      /*   Tag identifier.    */


GpiQueryTag(hps,
            &lTag);
