GpiCorrelateFrom - Syntax

This function performs a correlate operation on a section of the retained segment chain. 

#define INCL_GPICORRELATION /* Or use INCL_GPI, INCL_PM, */
#include <os2.h>

HPS        hps;            /*  Presentation-space handle. */
LONG       lFirstSegment;  /*  Specifies the first segment to be correlated. */
LONG       lLastSegment;   /*  Specifies the last segment to be correlated. */
LONG       lType;          /*  Type of segments on which correlation is to be performed. */
PPOINTL    pptlPick;       /*  Pick position. */
LONG       lMaxHits;       /*  Maximum hits. */
LONG       lMaxDepth;      /*  Number of pairs. */
PLONG      plSegTag;       /*  Segment identifiers and tags. */
LONG       lNumHits;       /*  Number of hits and error indicators. */

lNumHits = GpiCorrelateFrom(hps, lFirstSegment,
             lLastSegment, lType, pptlPick,
             lMaxHits, lMaxDepth, plSegTag);
