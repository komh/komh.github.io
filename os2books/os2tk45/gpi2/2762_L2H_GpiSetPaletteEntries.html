GpiSetPaletteEntries - Syntax

This function changes the entries in a palette. 

#define INCL_GPILOGCOLORTABLE /* Or use INCL_GPI, INCL_PM, */
#include <os2.h>

HPAL      hpal;      /*  Palette handle. */
ULONG     ulFormat;  /*  Format of entries in the table. */
ULONG     ulStart;   /*  Starting index. */
ULONG     ulCount;   /*  Count of elements in aulTable. */
PULONG    aulTable;  /*  Start of the application data area. */
BOOL      rc;        /*  Success indicator. */

rc = GpiSetPaletteEntries(hpal, ulFormat,
       ulStart, ulCount, aulTable);
