GpiBeginPath - Syntax

This function specifies the start of a path. 

#define INCL_GPIPATHS /* Or use INCL_GPI, INCL_PM, */
#include <os2.h>

HPS     hps;    /*  Presentation-space handle. */
LONG    lPath;  /*  Path identifier. */
BOOL    rc;     /*  Success indicator. */

rc = GpiBeginPath(hps, lPath);
