GpiIntersectClipRectangle - Syntax

This function sets the new clip region to the intersection of the current clip region and the specified rectangle. 

#define INCL_GPIREGIONS /* Or use INCL_GPI, INCL_PM, */
#include <os2.h>

HPS       hps;            /*  Presentation-space handle. */
PRECTL    prclRectangle;  /*  prclRectangle, the coordinates of which are world coordinates. */
LONG      lComplexity;    /*  Complexity of clipping and error indicators. */

lComplexity = GpiIntersectClipRectangle(hps,
                prclRectangle);
