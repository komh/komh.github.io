GpiQueryDeviceBitmapFormats - Syntax

This function returns the formats of bit maps supported internally by the device driver. 

#define INCL_GPIBITMAPS /* Or use INCL_GPI, INCL_PM, */
#include <os2.h>

HPS      hps;      /*  Presentation-space handle. */
LONG     lCount;   /*  Number of elements. */
PLONG    alArray;  /*  Data array. */
BOOL     rc;       /*  Success indicator. */

rc = GpiQueryDeviceBitmapFormats(hps, lCount,
       alArray);
