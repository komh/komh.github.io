GpiConvert - Syntax

This function converts an array of coordinate pairs from one coordinate space to another. 

#define INCL_GPITRANSFORMS /* Or use INCL_GPI, INCL_PM, */
#include <os2.h>

HPS        hps;         /*  Presentation-space handle. */
LONG       lSrc;        /*  Source coordinate space. */
LONG       lTarg;       /*  Target coordinate space. */
LONG       lCount;      /*  Number of coordinate pairs in aptlPoints. Must be greater or equal to 0. */
PPOINTL    aptlPoints;  /*  Array of coordinate pair structures. */
BOOL       rc;          /*  Success indicator. */

rc = GpiConvert(hps, lSrc, lTarg, lCount,
       aptlPoints);
