GpiQueryWidthTable - Syntax

This function returns width table information for the logical font identified by the value of the character-set attribute. 

#define INCL_GPILCIDS /* Or use INCL_GPI, INCL_PM, */
#include <os2.h>

HPS      hps;         /*  Presentation-space handle. */
LONG     lFirstChar;  /*  Codepoint of first character. */
LONG     lCount;      /*  Count of elements in alData. */
PLONG    alData;      /*  Array of width values. */
BOOL     rc;          /*  Success indicator. */

rc = GpiQueryWidthTable(hps, lFirstChar, lCount,
       alData);
