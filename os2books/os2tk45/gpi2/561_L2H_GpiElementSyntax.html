GpiElement - Syntax

This function adds a single element to the current segment. 

#define INCL_GPISEGEDITING /* Or use INCL_GPI, INCL_PM, */
#include <os2.h>

HPS      hps;      /*  Presentation-space handle. */
LONG     lType;    /*  Type to be associated with the element. */
PSZ      pszDesc;  /*  Element description. */
LONG     lLength;  /*  Length of content data for the element. */
PBYTE    pbData;   /*  Buffer pointer. */
LONG     lHits;    /*  Correlation and error indicators. */

lHits = GpiElement(hps, lType, pszDesc, lLength,
          pbData);
