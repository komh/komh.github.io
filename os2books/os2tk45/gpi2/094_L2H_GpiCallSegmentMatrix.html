GpiCallSegmentMatrix - Syntax

This function calls a segment and applies an instance transform to it. 

#define INCL_GPITRANSFORMS /* Or use INCL_GPI, INCL_PM, */
#include <os2.h>

HPS          hps;          /*  Presentation-space handle. */
LONG         lSegment;     /*  Identifier of segment to be called. */
LONG         lCount;       /*  Number of elements. */
PMATRIXLF    pmatlfArray;  /*  Instance transform matrix. */
LONG         lOptions;     /*  Transformation options. */
LONG         lHits;        /*  Correlation and error indicators. */

lHits = GpiCallSegmentMatrix(hps, lSegment,
          lCount, pmatlfArray, lOptions);
