GpiDrawBits - Syntax

This function draws a rectangle of bits. 

#define INCL_GPIBITMAPS /* Or use INCL_GPI, INCL_PM, Also in COMMON section */
#include <os2.h>

HPS             hps;            /*  Target presentation-space handle. */
PVOID           pBits;          /*  Source bits. */
PBITMAPINFO2    pbmiInfoTable;  /*  Bit-map information table. */
LONG            lCount;         /*  Point count. */
PPOINTL         aptlPoints;     /*  Point array. */
LONG            lRop;           /*  Mixing function required. */
ULONG           flOptions;      /*  Options. */
LONG            lHits;          /*  Correlation and error indicators. */

lHits = GpiDrawBits(hps, pBits, pbmiInfoTable,
          lCount, aptlPoints, lRop, flOptions);
