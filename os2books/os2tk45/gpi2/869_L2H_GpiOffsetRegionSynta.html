GpiOffsetRegion - Syntax

This function moves a region. 

#define INCL_GPIREGIONS /* Or use INCL_GPI, INCL_PM, */
#include <os2.h>

HPS        hps;         /*  Presentation-space handle. */
HRGN       Hrgn;        /*  Handle of the region to be moved. */
PPOINTL    pptlOffset;  /*  Offset to be added to the region boundary. */
BOOL       rc;          /*  Success indicator. */

rc = GpiOffsetRegion(hps, Hrgn, pptlOffset);
