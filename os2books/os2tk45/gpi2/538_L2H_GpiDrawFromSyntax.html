GpiDrawFrom - Syntax

This function draws a section of the segment chain. 

#define INCL_GPISEGMENTS /* Or use INCL_GPI, INCL_PM, */
#include <os2.h>

HPS     hps;            /*  Presentation-space handle. */
LONG    lFirstSegment;  /*  First segment to be drawn; it must be greater than 0. */
LONG    lLastSegment;   /*  Last segment to be drawn; it must be greater than 0. */
BOOL    rc;             /*  Success indicator. */

rc = GpiDrawFrom(hps, lFirstSegment, lLastSegment);
