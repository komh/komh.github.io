GpiDeleteElementRange - Syntax

This function deletes all elements between, and including, the elements indicated by the specified element numbers. 

#define INCL_GPISEGEDITING /* Or use INCL_GPI, INCL_PM, */
#include <os2.h>

HPS     hps;            /*  Presentation-space handle. */
LONG    lFirstElement;  /*  Number of the first element to be deleted. */
LONG    lLastElement;   /*  Number of the last element to be deleted. */
BOOL    rc;             /*  Success indicator. */

rc = GpiDeleteElementRange(hps, lFirstElement,
       lLastElement);
