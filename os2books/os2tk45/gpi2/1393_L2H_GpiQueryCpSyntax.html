GpiQueryCp - Syntax

This function returns the currently selected graphics code page. 

#define INCL_GPILCIDS /* Or use INCL_GPI, INCL_PM, */
#include <os2.h>

HPS      hps;         /*  Presentation-space handle. */
ULONG    ulCodePage;  /*  Code page. */

ulCodePage = GpiQueryCp(hps);
