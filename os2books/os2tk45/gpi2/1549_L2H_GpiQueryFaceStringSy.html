GpiQueryFaceString - Syntax

This function generates a compound face name for a font. 

#define INCL_GPILCIDS /* Or use INCL_GPI, INCL_PM, */
#include <os2.h>

HPS              ps;                /*  Presentation-space handle. */
PSZ              FamilyName;        /*  Family name. */
PFACENAMEDESC    attrs;             /*  Face-name description. */
LONG             length;            /*  Length of CompoundFaceName buffer. */
PSZ              CompoundFaceName;  /*  Compound face name. */
ULONG            cbRetLength;       /*  Length of the compound face name. */

cbRetLength = GpiQueryFaceString(ps, FamilyName,
                attrs, length, CompoundFaceName);
