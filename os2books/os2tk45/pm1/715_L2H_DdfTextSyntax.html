DdfText - Syntax

This function adds text to the DDF buffer. 

#define INCL_DDF
#include <os2.h>

HDDF    hddf;     /*  Handle to DDF returned by DdfInitialize. */
PCSZ    pszText;  /*  Pointer to the text buffer to be formatted. */
BOOL    rc;       /*  Success indicator. */

rc = DdfText(hddf, pszText);
