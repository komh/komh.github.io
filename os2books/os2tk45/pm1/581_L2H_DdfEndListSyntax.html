DdfEndList - Syntax

This function terminates the definition list initialized by DdfBeginList. 

#define INCL_DDF
#include <os2.h>

HDDF    hddf;  /*  Handle to DDF returned by DdfInitialize. */
BOOL    rc;    /*  Success indicator. */

rc = DdfEndList(hddf);
