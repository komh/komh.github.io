DosDelete - Syntax

Removes a file name from a directory. The deleted file may be recoverable. 

#define INCL_DOSFILEMGR
#include <os2.h>

PSZ       pszFile;  /*  Address of the name of the file to be deleted. */
APIRET    ulrc;     /*  Return Code. */

ulrc = DosDelete(pszFile);
