KbdGetStatus - Syntax

Returns information about the keyboard. 

#define INCL_KBD
#include <os2.h>

PKBDINFO    pkbdinfo;  /*  Pointer to keyboard data. */
HKBD        hkbd;      /*  Reserved.  Must be 0. */
APIRET      rc;        /*  Return code. */

rc = KbdGetStatus(pkbdinfo, hkbd);
