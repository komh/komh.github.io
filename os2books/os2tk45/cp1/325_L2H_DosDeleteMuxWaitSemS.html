DosDeleteMuxWaitSem - Syntax

Deletes an event semaphore or a mutex semaphore from a muxwait-semaphore list. 

#define INCL_DOSSEMAPHORES
#include <os2.h>

HMUX      hmux;  /*  The handle of the muxwait semaphore that is to have a semaphore deleted from its semaphore record list. */
HSEM      hSem;  /*  The handle of the semaphore that is to be deleted from the semaphore record list of the muxwait semaphore. */
APIRET    ulrc;  /*  Return Code. */

ulrc = DosDeleteMuxWaitSem(hmux, hSem);
