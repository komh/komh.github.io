VioPopUp - Syntax

Displays a temporary screen with a message to the user. 

#define INCL_VIO
#include <os2.h>

PULONG    Options;    /*  Option flags. */
HVIO      VioHandle;  /*  Presentation-space handle. */
APIRET    rc;         /*  Return code. */

rc = VioPopUp(Options, VioHandle);
