DosTmrQueryFreq - Syntax

Provides the frequency of the IRQ0 high resolution timer (Intel 8254). 

#define INCL_DOSPROFILE
#include <os2.h>

PULONG    pulTmrFreq;  /*  Timer Frequency. */
APIRET    rc;          /*  Return Code. */

rc = DosTmrQueryFreq(pulTmrFreq);
