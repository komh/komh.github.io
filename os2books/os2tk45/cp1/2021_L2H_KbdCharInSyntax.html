KbdCharIn - Syntax

Reads a character data record from the keyboard. 

#define INCL_KBD
#include <os2.h>

PKBDKEYINFO    CharData;  /*  Pointer to character data. */
ULONG          Wait;      /*  Wait Flag. */
HKBD           hkbd;      /*  Reserved.  Must be 0. */
APIRET         rc;        /*  Return code. */

rc = KbdCharIn(CharData, Wait, hkbd);
