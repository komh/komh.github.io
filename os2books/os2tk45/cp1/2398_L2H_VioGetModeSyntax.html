VioGetMode - Syntax

Returns the mode of the display. 

#define INCL_VIO
#include <os2.h>

PVIOMODEINFO    ModeData;   /*  Mode characteristics. */
HVIO            VioHandle;  /*  VIO presentation-space handle. */
APIRET          rc;         /*  Return code. */

rc = VioGetMode(ModeData, VioHandle);
