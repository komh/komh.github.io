DosCloseVDD - Syntax

Closes the specified virtual device driver (VDD) handle. 

#define INCL_DOSMVDM
#include <os2.h>

HVDD      hvdd;  /*  The handle of the virtual device driver to be closed. */
APIRET    ulrc;  /*  Return Code. */

ulrc = DosCloseVDD(hvdd);
