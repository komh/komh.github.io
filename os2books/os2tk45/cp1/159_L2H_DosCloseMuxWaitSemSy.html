DosCloseMuxWaitSem - Syntax

Closes a muxwait semaphore. 

#define INCL_DOSSEMAPHORES
#include <os2.h>

HMUX      hmux;  /*  The handle of the muxwait semaphore to close. */
APIRET    ulrc;  /*  Return Code. */

ulrc = DosCloseMuxWaitSem(hmux);
