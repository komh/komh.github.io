DosFindClose - Syntax

Closes the handle to a find request; that is, ends a search. 

#define INCL_DOSFILEMGR
#include <os2.h>

HDIR      hDir;  /*  Directory search handle. */
APIRET    ulrc;  /*  Return Code. */

ulrc = DosFindClose(hDir);
