DosWaitMuxWaitSem - Syntax

Waits for a muxwait semaphore to clear. 

#define INCL_DOSSEMAPHORES
#include <os2.h>

HMUX      hmux;       /*  The handle of the muxwait semaphore to wait for. */
ULONG     ulTimeout;  /*  The time-out in milliseconds. */
PULONG    pulUser;    /*  A pointer to a ULONG in which a user-defined value is returned. */
APIRET    ulrc;       /*  Return Code. */

ulrc = DosWaitMuxWaitSem(hmux, ulTimeout,
         pulUser);
