KbdSetStatus - Syntax

Sets the characteristics of the keyboard. 

#define INCL_KBD
#include <os2.h>

PKBDINFO    pkbdinfo;  /*  Pointer to keyboard status. */
HKBD        hkbd;      /*  Reserved.  Must be 0. */
APIRET      rc;        /*  Return code. */

rc = KbdSetStatus(pkbdinfo, hkbd);
