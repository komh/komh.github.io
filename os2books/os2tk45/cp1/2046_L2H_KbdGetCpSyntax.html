KbdGetCp - Syntax

Gets the current keyboard code page. 

#define INCL_KBD
#include <os2.h>

ULONG      ulReserved;  /*  Reserved.  Must be 0. */
PUSHORT    pidCP;       /*  Pointer to Code-page ID. */
HKBD       hkbd;        /*  Reserved.  Must be 0. */
APIRET     return;      /*  Return code. */

return = KbdGetCp(ulReserved, pidCP, hkbd);
