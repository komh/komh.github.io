DosSetDefaultDisk - Syntax

Sets the specified drive as the default drive. 

#define INCL_DOSFILEMGR
#include <os2.h>

ULONG     disknum;  /*  New default-drive number. */
APIRET    ulrc;     /*  Return Code. */

ulrc = DosSetDefaultDisk(disknum);
