DosCloseMutexSem - Syntax

Closes a mutex semaphore. 

#define INCL_DOSSEMAPHORES
#include <os2.h>

HMTX      hmtx;  /*  The handle of the mutex semaphore to close. */
APIRET    ulrc;  /*  Return Code. */

ulrc = DosCloseMutexSem(hmtx);
