VioGetDeviceCellSize - Syntax

Returns the size of the current character cell in pels. 

#define INCL_VIO
#include <os2.h>

PULONG    Height;
PULONG    Width;
HVIO      hvps;    /*  VIO presentation-space handle. */
APIRET    rc;      /*  Return code. */

rc = VioGetDeviceCellSize(Height, Width, hvps);
