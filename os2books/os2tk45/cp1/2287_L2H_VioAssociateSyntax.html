VioAssociate - Syntax

Associates or disassociates a VIO presentation space with a device context. 

#define INCL_VIO
#include <os2.h>

HDC       hdc;   /*  Device-context handle. */
HVIO      hvps;  /*  VIO presentation space handle. */
APIRET    rc;    /*  Return code. */

rc = VioAssociate(hdc, hvps);
