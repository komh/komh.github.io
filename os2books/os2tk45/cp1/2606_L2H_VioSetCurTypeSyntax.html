VioSetCurType - Syntax

Sets the cursor type. 

#define INCL_VIO
#include <os2.h>

PVIOCURSORINFO    CursorData;  /*  Cursor characteristics. */
HVIO              VioHandle;   /*  VIO presentation-space handle. */
APIRET            rc;          /*  Return code. */

rc = VioSetCurType(CursorData, VioHandle);
