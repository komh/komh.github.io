DosProtectSetFHState - Syntax

Sets the state of the specified protected file handle. 

#define INCL_DOSFILEMGR
#include <os2.h>

HFILE     hFile;               /*  File handle to be set. */
ULONG     mode;                /*  Contents of the fsOpenMode field defined in a previous DosOpen. */
FHLOCK    fhFileHandleLockID;  /*  The lockid of the protected file handle. */
APIRET    ulrc;                /*  Return Code. */

ulrc = DosProtectSetFHState(hFile, mode, fhFileHandleLockID);
