KbdFlushBuffer - Syntax

Removes all entries from the keyboard buffer. This discards all user keystrokes typed ahead. 

#define INCL_KBD
#include <os2.h>

HKBD      hkbd;    /*  Reserved.  Must be 0. */
APIRET    return;  /*  Return code. */

return = KbdFlushBuffer(hkbd);
