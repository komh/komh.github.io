VioScrUnLock - Syntax

Releases ownership of (unlocks) the physical display buffer. 

#define INCL_VIO
#include <os2.h>

HVIO      VioHandle;  /*  VIO presentation-space handle. */
APIRET    rc;         /*  Return code. */

rc = VioScrUnLock(VioHandle);
