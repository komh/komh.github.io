KbdXlate - Syntax

Translates scan codes with shift states into ASCII codes. 

#define INCL_KBD
#include <os2.h>

PKBDKEYINFO    pKey;  /*  Pointer to character data. */
HKBD           hkbd;  /*  Reserved.  Must be 0. */
APIRET         rc;    /*  Return code. */

rc = KbdXlate(pKey, hkbd);
