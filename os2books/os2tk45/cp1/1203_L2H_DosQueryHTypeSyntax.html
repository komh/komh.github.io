DosQueryHType - Syntax

Determines whether a handle refers to a file or a device. 

#define INCL_DOSFILEMGR
#include <os2.h>

HFILE     hFile;  /*  File handle. */
PULONG    pType;  /*  Address of the value indicating the handle type. */
PULONG    pAttr;  /*  Address of the device-driver attribute word if pType indicates a local character device. */
APIRET    ulrc;   /*  Return Code. */

ulrc = DosQueryHType(hFile, pType, pAttr);
