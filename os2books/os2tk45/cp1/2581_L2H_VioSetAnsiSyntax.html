VioSetAnsi - Syntax

Activates or deactivates ANSI support. 

#define INCL_VIO
#include <os2.h>

ULONG     Indicator;  /*  On/Off indicator. */
HVIO      VioHandle;  /*  VIO presentation-space handle. */
APIRET    rc;         /*  Return code. */

rc = VioSetAnsi(Indicator, VioHandle);
