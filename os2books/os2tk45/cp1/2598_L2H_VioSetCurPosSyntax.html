VioSetCurPos - Syntax

Sets the cursor's coordinates on the screen. 

#define INCL_VIO
#include <os2.h>

ULONG     Row;
USHORT    Column;
HVIO      VioHandle;  /*  VIO presentation-space handle. */
APIRET    rc;         /*  Return code. */

rc = VioSetCurPos(Row, Column, VioHandle);
