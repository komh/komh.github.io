MouFlushQue - Syntax

Directs the mouse device driver to empty the mouse-event queue. 

#define INCL_MOU
#include <os2.h>

HMOU      DeviceHandle;  /*  Mouse device handle. */
APIRET    rc;            /*  Return code. */

rc = MouFlushQue(DeviceHandle);
