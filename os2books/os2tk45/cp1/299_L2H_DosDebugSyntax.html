DosDebug - Syntax

Enables the calling application to control another application for debugging purposes. 

#define INCL_DOSPROCESS
#include <os2.h>

PVOID     pdbgbuf;  /*  Address of a DosDebug Buffer. */
APIRET    ulrc;     /*  Return Code. */

ulrc = DosDebug(pdbgbuf);
