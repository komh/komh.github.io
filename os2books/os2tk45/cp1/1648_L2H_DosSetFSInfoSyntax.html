DosSetFSInfo - Syntax

Sets information for a file system device. 

#define INCL_DOSFILEMGR
#include <os2.h>

ULONG     disknum;
ULONG     infolevel;
PVOID     pBuf;
ULONG     cbBuf;
APIRET    ulrc;       /*  Return Code. */

ulrc = DosSetFSInfo(disknum, infolevel, pBuf,
         cbBuf);
