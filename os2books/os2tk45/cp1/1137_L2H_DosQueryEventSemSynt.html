DosQueryEventSem - Syntax

Retrieves the post count for an event semaphore. 

#define INCL_DOSSEMAPHORES
#include <os2.h>

HEV       hev;        /*  The handle of the event semaphore to query. */
PULONG    pulPostCt;  /*  A pointer to the semaphore's post count. */
APIRET    ulrc;       /*  Return Code. */

ulrc = DosQueryEventSem(hev, pulPostCt);
