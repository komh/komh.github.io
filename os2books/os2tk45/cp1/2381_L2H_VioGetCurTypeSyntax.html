VioGetCurType - Syntax

Returns the cursor type. 

#define INCL_VIO
#include <os2.h>

PVIOCURSORINFO    CursorData;  /*  Cursor characteristics. */
HVIO              VioHandle;   /*  Presentation-space handle. */
APIRET            rc;          /*  Return code. */

rc = VioGetCurType(CursorData, VioHandle);
