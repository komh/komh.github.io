DosSubUnsetMem - Syntax

Ends the use of a memory pool. 

#define INCL_DOSMEMMGR
#include <os2.h>

PVOID     pbBase;  /*  The offset of the memory pool whose use is being terminated. */
APIRET    ulrc;    /*  Return Code. */

ulrc = DosSubUnsetMem(pbBase);
