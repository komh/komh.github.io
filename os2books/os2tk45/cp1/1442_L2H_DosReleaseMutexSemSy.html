DosReleaseMutexSem - Syntax

Relinquishes ownership of a mutex semaphore. 

#define INCL_DOSSEMAPHORES
#include <os2.h>

HMTX      hmtx;  /*  The handle of the mutex semaphore to release. */
APIRET    ulrc;  /*  Return Code. */

ulrc = DosReleaseMutexSem(hmtx);
