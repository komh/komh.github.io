KbdSetLayout - Syntax

Allows the keyboard layout to be changed. This change affects only the current session. 

#define INCL_KBD
#include <os2.h>

PSZ       pszName;  /*  Keyboard layout name. */
HKBD      hkbd;     /*  Reserved.  Must be zero. */
APIRET    rc;       /*  Return code. */

rc = KbdSetLayout(pszName, hkbd);
