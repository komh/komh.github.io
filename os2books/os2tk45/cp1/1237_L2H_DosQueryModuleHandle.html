DosQueryModuleHandle - Syntax

Returns the handle of a dynamic link module that was previously loaded. 

#define INCL_DOSMODULEMGR
#include <os2.h>

PSZ         pszModname;
PHMODULE    phmod;
APIRET      ulrc;        /*  Return Code. */

ulrc = DosQueryModuleHandle(pszModname, phmod);
