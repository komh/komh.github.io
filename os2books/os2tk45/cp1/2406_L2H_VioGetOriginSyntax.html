VioGetOrigin - Syntax

Gets the position at which the presentation space maps to the window. 

#define INCL_VIO
#include <os2.h>

PULONG    Row;
PULONG    Column;
HVIO      hvps;    /*  VIO presentation-space handle. */
APIRET    rc;      /*  Return code. */

rc = VioGetOrigin(Row, Column, hvps);
