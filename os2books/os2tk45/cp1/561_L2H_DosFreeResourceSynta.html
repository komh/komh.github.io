DosFreeResource - Syntax

Frees a resource that was loaded by DosGetResource. 

#define INCL_DOSMODULEMGR
#include <os2.h>

PVOID     pb;    /*  The address of the resource to be freed. */
APIRET    ulrc;  /*  Return Code. */

ulrc = DosFreeResource(pb);
