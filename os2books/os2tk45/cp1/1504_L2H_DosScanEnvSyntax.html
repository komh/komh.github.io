DosScanEnv - Syntax

Searches an environment segment for an environment variable. 

#define INCL_DOSFILEMGR
#define INCL_DOSMISC
#include <os2.h>

PSZ       pszName;    /*  Address of the name of the environment variable. */
PSZ      *ppszValue;  /*  A pointer to the PSZ in which a pointer to the environment string is returned by the system. */
APIRET    ulrc;       /*  Return Code. */

ulrc = DosScanEnv(pszName, ppszValue);
