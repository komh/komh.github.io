DosClose - Syntax

Closes a handle to a file, pipe, or device. 

#define INCL_DOSFILEMGR
#include <os2.h>

HFILE     hFile;  /*  The handle returned by a previous call to DosCreateNPipe, DosCreatePipe, DosDupHandle, or DosOpen. */
APIRET    ulrc;   /*  Return Code. */

ulrc = DosClose(hFile);
