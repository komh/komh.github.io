VioGetAnsi - Syntax

Returns the current ANSI status on and off state. 

#define INCL_VIO
#include <os2.h>

PULONG    Indicator;  /*  Address of the current ANSI status. */
HVIO      VioHandle;  /*  Presentation-space handle. */
APIRET    rc;         /*  Return code. */

rc = VioGetAnsi(Indicator, VioHandle);
