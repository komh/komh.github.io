DosError - Syntax

Disables or enables error notification to end users for errors in the current process. 

#define INCL_DOSMISC
#include <os2.h>

ULONG     error;  /*  Error and Exception pop-up flags. */
APIRET    ulrc;   /*  Return Code. */

ulrc = DosError(error);
