DosQueryFHState - Syntax

Queries the state of the specified file handle. 

#define INCL_DOSFILEMGR
#include <os2.h>

HFILE     hFile;  /*  Handle of the file to be queried. */
PULONG    pMode;  /*  Address of the contents of the open-mode word defined in a previous DosOpen function. */
APIRET    ulrc;   /*  Return Code. */

ulrc = DosQueryFHState(hFile, pMode);
