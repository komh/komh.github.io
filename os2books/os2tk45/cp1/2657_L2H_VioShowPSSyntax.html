VioShowPS - Syntax

Updates the display of the VIO presentation space. 

#define INCL_VIO
#include <os2.h>

ULONG     Depth;  /*  Depth. */
ULONG     Width;  /*  Width. */
ULONG     Cell;   /*  Offset to first updated cell. */
HVIO      hvps;   /*  VIO presentation handle. */
APIRET    rc;     /*  Return code */

rc = VioShowPS(Depth, Width, Cell, hvps);
