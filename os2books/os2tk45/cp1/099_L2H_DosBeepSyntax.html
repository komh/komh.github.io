DosBeep - Syntax

Generates sound from the speaker. 

#define INCL_DOSPROCESS
#include <os2.h>

ULONG     freq;  /*  Cycles per second (Hertz) in the range of 0x25 to 0x7FFF. */
ULONG     dur;   /*  The length of the sound in milliseconds. */
APIRET    ulrc;  /*  Return Code. */

ulrc = DosBeep(freq, dur);
