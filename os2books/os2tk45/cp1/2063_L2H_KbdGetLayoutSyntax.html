KbdGetLayout - Syntax

Returns the name of the keyboard layout in use. 

#define INCL_KBD
#include <os2.h>

PSZ       pszName;  /*  Keyboard layout name. */
HKBD      hkbd;     /*  Reserved.  Must be 0. */
APIRET    rc;       /*  Return code. */

rc = KbdGetLayout(pszName, hkbd);
