DosQueryThreadContext - Syntax

Query context of a suspended thread. 

#define INCL_DOSEXCEPTIONS
#include <os2.h>

TID               tid;    /*  Thread identity. */
ULONG             level;  /*  Level of information desired. */
PCONTEXTRECORD    pcxt;   /*  Thread context record. */
APIRET            ulrc;   /*  Return Code. */

ulrc = DosQueryThreadContext(tid, level, pcxt);
