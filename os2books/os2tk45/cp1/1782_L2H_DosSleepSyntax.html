DosSleep - Syntax

Suspends the current thread for a specified time interval. 

#define INCL_DOSPROCESS
#include <os2.h>

ULONG     msec;  /*  The time, in milliseconds, for which the calling thread's execution will be suspended. */
APIRET    ulrc;  /*  Return Code. */

ulrc = DosSleep(msec);
