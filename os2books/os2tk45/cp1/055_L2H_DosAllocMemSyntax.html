DosAllocMem - Syntax

Allocates a private memory object within the virtual-address space. 

#define INCL_DOSMEMMGR
#include <os2.h>

PPVOID    ppb;   /*  A pointer to a variable that will receive the base address of the allocated private memory object. */
ULONG     cb;    /*  Size, in bytes, of the private memory object to allocate. */
ULONG     flag;  /*  Allocate attribute and desired access protection flags. */
APIRET    ulrc;  /*  Return Code. */

ulrc = DosAllocMem(ppb, cb, flag);
