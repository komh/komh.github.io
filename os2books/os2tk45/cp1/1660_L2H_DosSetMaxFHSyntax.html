DosSetMaxFH - Syntax

Defines the maximum number of file handles for the calling process. 

#define INCL_DOSFILEMGR
#include <os2.h>

ULONG     cFH;   /*  Total number of file handles to be provided. */
APIRET    ulrc;  /*  Return Code. */

ulrc = DosSetMaxFH(cFH);
