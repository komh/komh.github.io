DosSetFHState - Syntax

Sets the state of the specified file handle. 

#define INCL_DOSFILEMGR
#include <os2.h>

HFILE     hFile;  /*  File handle to be set. */
ULONG     mode;   /*  Contents of the fsOpenMode field defined in a previous DosOpen function. */
APIRET    ulrc;   /*  Return Code. */

ulrc = DosSetFHState(hFile, mode);
