KbdGetLayoutUni - Syntax

Returns the name of the keyboard layout in use. 

#define INCL_KBD
#include <os2.h>

USHORT     *name;  /*  Keyboard layout name. */
HKBD        hkbd;  /*  Reserved.  Must be 0. */
APIRET      rc;    /*  Return code. */

rc = KbdGetLayoutUni(name, hkbd);
