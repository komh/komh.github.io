DosCloseEventSem - Syntax

Closes an event semaphore. 

#define INCL_DOSSEMAPHORES
#include <os2.h>

HEV       hev;   /*  The handle of the event semaphore to close. */
APIRET    ulrc;  /*  Return Code. */

ulrc = DosCloseEventSem(hev);
