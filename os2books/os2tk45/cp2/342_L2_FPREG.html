FPREG

Coprocessor stack register element. 

typedef struct _FPREG {
  ULONG      losig;    /*  Low 32-bits of the significand. */
  ULONG      hisig;    /*  High 32-bits of the significand. */
  USHORT     signexp;  /*  Sign and exponent. */
} FPREG;

typedef   FPREG   * PFPREG ;
