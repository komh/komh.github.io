WinSetCp - Syntax

This function sets the code page for a queue. 

#define INCL_WINCOUNTRY /* Or use INCL_WIN, INCL_PM, */
#include <os2.h>

HMQ      hmq;         /*  Message-queue handle. */
ULONG    ulCodePage;  /*  Code page. */
BOOL     rc;          /*  Success indicator. */

rc = WinSetCp(hmq, ulCodePage);
