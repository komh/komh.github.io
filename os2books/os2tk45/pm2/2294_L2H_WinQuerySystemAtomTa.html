WinQuerySystemAtomTable - Example Code

This function queries the length of an atom. 

#define INCL_WINATOM
#include <OS2.H>

HATOMTBL      hatomtbl;
ATOM          atom;
unsigned char szAtomName;

hatomtbl = WinQuerySystemAtomTable();

atom = WinFindAtom(hatomtbl, &szAtomName);
