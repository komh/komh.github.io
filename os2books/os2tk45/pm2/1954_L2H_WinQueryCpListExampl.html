WinQueryCpList - Example Code

This example queries available code pages. 

#define INCL_WINCOUNTRY
#include <OS2.H>
#define maxcount 8
HAB hab;
/* . */
ULONG  aulCodepage[maxcount];


WinQueryCpList(hab,
               (ULONG)maxcount,
               (PULONG) aulCodepage);
