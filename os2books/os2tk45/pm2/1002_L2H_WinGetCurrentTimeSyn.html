WinGetCurrentTime - Syntax

This function returns the current time. 

#define INCL_WINTIMER /* Or use INCL_WIN, INCL_PM, */
#include <os2.h>

HAB      hab;     /*  Anchor-block handle. */
ULONG    ulTime;  /*  System-timer count. */

ulTime = WinGetCurrentTime(hab);
