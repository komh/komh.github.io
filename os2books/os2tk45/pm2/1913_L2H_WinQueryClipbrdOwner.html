WinQueryClipbrdOwner - Example Code

This example finds out which window currently owns the clipboard. 

#define INCL_WINCLIPBOARD
#include <OS2.H>
HAB hab;
/* . */
HWND  hwndClipbrdOwner;


hwndClipbrdOwner = WinQueryClipbrdOwner(hab);
