WinQueryClipbrdData - Example Code

This example obtains a handle to the current clipboard data of text format. 

#define INCL_WINCLIPBOARD
#include <OS2.H>
HAB hab;
/* . */
ULONG hclipbrdData;

hclipbrdData = WinQueryClipbrdData(hab, CF_TEXT);
