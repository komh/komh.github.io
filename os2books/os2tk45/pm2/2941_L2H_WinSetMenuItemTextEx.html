WinSetMenuItemText - Example Code

This example sets the options text in a menu. 

#define INCL_WINWINDOWMGR
#include <OS2.H>
#define IDM_OPTIONS 900

HWND    hwndMenu;

WinSetMenuItemText(hwndMenu,
                        IDM_OPTIONS,
                        "Options");
