WinOpenClipbrd - Syntax

This function opens the clipboard. 

#define INCL_WINCLIPBOARD /* Or use INCL_WIN, INCL_PM, */
#include <os2.h>

HAB     hab;  /*  Anchor-block handle. */
BOOL    rc;   /*  Success indicator. */

rc = WinOpenClipbrd(hab);
