WinQueryAtomLength - Example Code

This function queries the length of an atom. 

#define INCL_WINATOM
#include <OS2.H>

HATOMTBL atomtbl;
ATOM atom = 25;

      WinQueryAtomlength(atomtbl,  /* atom handle. */
                         atom);
