WinQueryAtomName - Example Code

This function obtains the name of an atom given the atom id. 

#define INCL_WINATOM
#include <OS2.H>
HATOMTBL atomtbl;
char atomname[256];
ATOM atom = 25;

         WinQueryAtomName(atomtbl,
                         atom,
                         atomname,
                         sizeof(atomname));
