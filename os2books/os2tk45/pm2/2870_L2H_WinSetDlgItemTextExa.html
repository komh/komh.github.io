WinSetDlgItemText - Example Code

This example sets the text "CALENDAR" in a dialog box. 

#define INCL_WINDIALOGS
#include <OS2.H>
#define ID_DLG_CALENDAR 900
HWND    hwndDlg;

    WinSetDlgItemText(hwndDlg,
                      ID_DLG_CALENDAR,
                      "CALENDAR");
