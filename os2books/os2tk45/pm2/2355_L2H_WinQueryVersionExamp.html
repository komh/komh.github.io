WinQueryVersion - Example Code

This example gets the version of PM that is running. 

#define INCL_WINWINDOWMGR
#include <OS2.H>
#include <stdio.h>

HAB   hab;
ULONG lVer;

lVer = WinQueryVersion(hab);
printf("PM revision is %d ", lVer);
