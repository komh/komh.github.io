WinQueryCp - Syntax

This function returns the queue code page for the specified message queue. 

#define INCL_WINCOUNTRY /* Or use INCL_WIN, INCL_PM, */
#include <os2.h>

HMQ      hmq;         /*  Message queue. */
ULONG    ulCodePage;  /*  Code page. */

ulCodePage = WinQueryCp(hmq);
