WinQueryWindowDC - Example Code

This example shows how to check if WinOpenWindowDC has been called for this window. 

#define INCL_WINWINDOWMGR
#include <OS2.H>

HWND  hwndClient;           /* window handle. */

if(WinQueryWindowDC(hwndClient))
{
     /*  ...    */
}
