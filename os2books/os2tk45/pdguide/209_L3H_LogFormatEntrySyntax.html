LogFormatEntry - Syntax

LogFormatEntry formats a Log Entry for display. 

#define INCL_LOGGING
#include <os2.h>
#include <lfdef.h>

ULONG     service;
PVOID     pFormatEntry;
APIRET    rc;

rc = LogFormatEntry(service, pFormatEntry);
