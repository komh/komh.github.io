LogCloseFile - Syntax

LogCloseFile closes a file that was previously opened by LogOpenFile. 

#define INCL_LOGGING
#include <os2.h>
#include <lfdef.h>

ULONG     service;
PVOID     pCloseFile;
APIRET    rc;

rc = LogCloseFile(service, pCloseFile);
