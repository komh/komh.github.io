somGetSize - Example Code

#include <animal.h>
void main()
{
   Animal myAnimal;
   long animalSize;
   myAnimal = AnimalNew();
   animalSize = _somGetSize(myAnimal);
   somPrintf("Size of animal (in bytes): %d\n", animalSize);
   _somFree(myAnimal);
}
/*
Output from this program:
Size of animal (in bytes): 8
*/
