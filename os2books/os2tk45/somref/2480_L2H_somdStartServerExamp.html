somdStartServer - Example Code

#include <somd.h>
#include <servmgr.h>

SOMDServerMgr servmgr;
string  server_alias = "MyServer";
ORBStatus rc;
Environment e;

SOM_InitEnvironment(&e);
SOMD_Init(&e);
servmgr = SOMDServerMgrNew();
rc = _somdStartServer(servmgr, &e, server_alias);
