somGetVersionNumbers - Example Code

#include <som.h>

main() {

  long major, minor;
  SOMClass myClass;

  somEnvironmentNew();
  myClass = _somFindClass(SOMClassMgrObject,
                       somIdFromString("Animal"), 0, 0);
  _somGetVersionNumbers(myClass, &major, &minor);
  somPrintf("The version numbers are %i and %i.\n", major, minor);
}
