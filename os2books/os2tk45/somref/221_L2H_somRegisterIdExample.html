somRegisterId - Example Code

#include <som.h>
static string s = "unregistered";
static somId sid = &s;
main()
{
   somEnvironmentNew();
   SOM_Test(somRegisterId(sid) == 1);
   SOM_Test(somRegisterId(somIdFromString("registered")) == 0);
}
