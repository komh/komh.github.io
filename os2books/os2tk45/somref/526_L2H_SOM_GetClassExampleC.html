SOM_GetClass - Example Code

#include <somcls.xh>
#include <animal.xh>
main()
{
   Animal *a = new Animal;
   SOMClass cls1 = SOM_GetClass(a);
   SOMClass cls2 = a->somGetClass();
   if (cls1 == cls2)
      printf("macro and method for getClass the same for Animal\n");
   else
      printf("macro and method for getClass not same for Animal\n");
}
