somIsObj - Example Code

#include <stdio.h>
#include <som.xh>

void example(void *memPtr)
{
   if (!somIsObj(memPtr))
     printf("memPtr is not a valid SOM object.\n");
   else
     printf("memPtr points to an object of class %s\n",
             ((SOMObject *)memPtr)->somGetClassName());
}
