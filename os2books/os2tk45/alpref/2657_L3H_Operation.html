Operation

(* ROL - Rotate Left *)
temp  COUNT;
WHILE (temp <> 0)
DO
tmpcf  high-order bit of (r/m);
r/m  r/m * 2 + (tmpcf);
temp  temp - 1;
OD;
IF COUNT = 1
THEN
IF high-order bit of r/m <> CF
THEN OF  1;
ELSE OF  0;
FI;
ELSE OF  undefined;
FI;
(* ROR - Rotate Right *)
temp  COUNT;
WHILE (temp <> 0)
DO
tmpcf  low-order bit of (r/m);
r/m  r/m / 2 + (tmpcf * 2widty(r/m));
temp  temp - 1;
DO;
IF COUNT = 1
THEN
IF (high-order bit of r/m) <> (bit next to high-order bit of r/m)
THEN OF  1;
ELSE OF  0;
FI;
ELSE OF  undefined;
FI;
