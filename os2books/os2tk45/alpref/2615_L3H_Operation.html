Operation

IF OperandSize = 16 (* instruction = POPA *)
THEN
DI  Pop();
SI  Pop();
BP  Pop();
increment SP by 2 (* skip next 2 bytes of stack *)
BX  Pop();
DX  Pop();
CX  Pop();
AX  Pop();
ELSE (* OperandSize = 32, instruction = POPAD *)
EDI  Pop();
ESI  Pop();
EBP  Pop();
increment SP by 4 (* skip next 4 bytes of stack *)
EBX  Pop();
EDX  Pop();
ECX  Pop();
EAX  Pop();
FI;
