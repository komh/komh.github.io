Operation

IF instruction = LIDT
THEN
IF OperandSize = 16
THEN IDTR.Limit:Base  m16:24 (* 24-bits of base loaded *)
ELSE IDTR.Limit:Base  m16:32
FI;
ELSE (* instruction = LGDT *)
IF OperandSize = 16
THEN GDTR.Limit:Base  m16:24 (* 24-bits of base loaded *)
ELSE GDTR.Limit:Base  m16:32;
FI;
FI;
