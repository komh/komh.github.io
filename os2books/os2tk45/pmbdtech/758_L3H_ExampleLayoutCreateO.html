Example - LayoutCreateObject

This example creates a layout object. 

#include <layout.h>

LAYOUT_OBJECT plh;
ULONG RC;

  RC= LayoutCreateObject (Locale_Arabic,&plh);   /* or: Locale_Hebrew */

  if (RC) { printf("Create Error! !!\n"); exit(0);}
