CTIME

CTIME Structure that contains time information for a data element in the details view of a container control. 
typedef struct _CTIME {
UCHAR    hours;       /* Hour  */
UCHAR    minutes;     /* Minute  */
UCHAR    seconds;     /* Second  */
UCHAR    ucReserved;  /* Reserved  */
 } CTIME;
