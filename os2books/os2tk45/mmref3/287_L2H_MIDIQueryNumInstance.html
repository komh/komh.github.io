MIDIQueryNumInstances - Syntax

This function returns the number of instances. 

#include <mididll.h>

PULONG    pulNumInstances;  /*  Number of instances. */
ULONG     ulFlag;           /*  Not used. */
ULONG     rc;               /*  Return code. */

rc = MIDIQueryNumInstances(pulNumInstances,
       ulFlag);
