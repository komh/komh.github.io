MIDITimer - Syntax

This function provides the ability to start or stop the RTMIDI timer. 

#include <mididll.h>

ULONG    ulAction;  /*  Start or stop. */
ULONG    ulFlag;    /*  Not used. */
ULONG    rc;        /*  Return code. */

rc = MIDITimer(ulAction, ulFlag);
