MIDIDeleteInstance - Syntax

This function deletes an instance. 

#include <mididll.h>

MINSTANCE    minstance;  /*  Instance number. */
ULONG        ulFlag;     /*  Not used. */
ULONG        rc;         /*  Return code. */

rc = MIDIDeleteInstance(minstance, ulFlag);
