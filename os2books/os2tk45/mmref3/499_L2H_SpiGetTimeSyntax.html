SpiGetTime - Syntax

This function queries the current stream time. 

#include <os2.h>

HSTREAM    hstream;  /*  Stream handle. */
PMMTIME    pmmtime;  /*  Current stream time. */
ULONG      rc;       /*  Return codes. */

rc = SpiGetTime(hstream, pmmtime);
