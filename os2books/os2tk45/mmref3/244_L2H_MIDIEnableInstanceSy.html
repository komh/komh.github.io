MIDIEnableInstance - Syntax

This function enables an instance for sending messages, receiving messages, or both. 

#include <mididll.h>

MINSTANCE    minstance;  /*  Instance number. */
ULONG        ulFlag;     /*  Flag. */
ULONG        rc;         /*  Return code. */

rc = MIDIEnableInstance(minstance, ulFlag);
