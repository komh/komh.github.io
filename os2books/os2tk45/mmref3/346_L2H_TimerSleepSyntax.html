TimerSleep - Syntax

This function blocks the current thread for the specified number of milliseconds. 

#include "mididll.h"

ULONG    ulDuration;  /*  Duration in ms. */
ULONG    ulFlag;      /*  Not used. */
ULONG    rc;          /*  Return code. */

rc = TimerSleep(ulDuration, ulFlag);
