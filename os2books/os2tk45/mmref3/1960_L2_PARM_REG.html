This structure contains fields for the SMH_REGISTER message. 

typedef struct _PARM_REG {
  ULONG         ulFunction;       /*  SMH command function. */

  HID FAR      *phidTgt;          /*  Handler ID of target. */
  PSHCFN        pshcfnEntry;      /*  Handler entry point. */

} PARM_REG;
