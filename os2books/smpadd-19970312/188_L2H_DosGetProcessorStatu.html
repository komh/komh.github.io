DosGetProcessorStatus - Syntax

Get the status for a specific processor. 

#define INCL_DOSSPINLOCK
#include <os2.h>

ULONG     ProcNum;  /*  Processor number. */
PULONG    pStatus;  /*  Pointer to the status returned for the specified processor. */
APIRET    ulrc;     /*  Return Code. */

ulrc = DosGetProcessorStatus(ProcNum, pStatus);
