DosDebug Buffer

DosDebug buffer structure. 

typedef struct _DosDebug Buffer {
  ULONG      Pid;     /*  Debuggee Process ID */
  ULONG      Tid;     /*  Debuggee Thread ID */
  LONG       Cmd;     /*  Command or Notification */
  LONG       Value;   /*  Generic Data Value */
  ULONG      Addr;    /*  Debuggee Address */
  ULONG      Buffer;  /*  Debugger Buffer Address */
  ULONG      Len;     /*  Length of Range */
  ULONG      Index;   /*  Generic Identifier Index */
  ULONG      MTE;     /*  Module Table Entry Handle */
  ULONG      EAX;     /*  Register Set */
  ULONG      ECX;
  ULONG      EDX;
  ULONG      EBX;
  ULONG      ESP;
  ULONG      EBP;
  ULONG      ESI;
  ULONG      EDI;
  ULONG      EFlags;
  ULONG      EIP;
  ULONG      CSLim;   /*  Byte Granular Limits */
  ULONG      CSBase;  /*  Byte Granular Base */
  UCHAR      CSAcc;   /*  Access Bytes */
  UCHAR      CSAtr;   /*  Attribute Bytes */
  USHORT     CS;
  ULONG      DSLim;
  ULONG      DSBase;
  UCHAR      DSAcc;
  UCHAR      DSAtr;
  USHORT     DS;
  ULONG      ESLim;
  ULONG      ESBase;
  UCHAR      ESAcc;
  UCHAR      ESAtr;
  USHORT     ES;
  ULONG      FSLim;
  ULONG      FSBase;
  UCHAR      FSAcc;
  UCHAR      FSAtr;
  USHORT     FS;
  ULONG      GSLim;
  ULONG      GSBase;
  UCHAR      GSAcc;
  UCHAR      GSAtr;
  USHORT     GS;
  ULONG      SSLim;
  ULONG      SSBase;
  UCHAR      SSAcc;
  UCHAR      SSAtr;
  USHORT     SS;
} DosDebug Buffer;
