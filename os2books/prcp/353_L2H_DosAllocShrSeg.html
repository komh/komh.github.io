EXTRN  DosAllocShrSeg:FAR
INCL_DOSMEMMGR      EQU 1

PUSH   WORD    Size          ;Number of bytes requested
PUSH@  ASCIIZ  Name          ;Name string

CALL   DosAllocShrSeg
