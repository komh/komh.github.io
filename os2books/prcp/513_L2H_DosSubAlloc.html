DosSubAlloc

EXTRN  DosSubAlloc:FAR
INCL_DOSMEMMGR      EQU 1

PUSH   WORD    SegSelector   ;Segment selector
PUSH@  WORD    BlockOffset   ;Block Offset (returned)
PUSH   WORD    Size          ;Size of requested block
CALL   DosSubAlloc

Returns WORD
