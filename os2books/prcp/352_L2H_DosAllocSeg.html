DosAllocSeg

EXTRN  DosAllocSeg:FAR
INCL_DOSMEMMGR      EQU 1

PUSH   WORD    Size          ;Number of bytes requested
PUSH@  WORD    Selector      ;Selector allocated (returned)
PUSH   WORD    AllocFlags    ;Allocation flags
CALL   DosAllocSeg

Returns WORD
