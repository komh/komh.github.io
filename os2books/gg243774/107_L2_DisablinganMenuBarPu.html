Disabling an Menu Bar/Pulldown Menu Item

hMenu = WinWindowFromID(hFrame,            /* Get menu bar handle        */
                      FID_MENU);

rc = WinEnableMenuItem(hMenu,              /* Menu bar handle            */
                       MI_VIEW,            /* Menu item identifier       */
                       TRUE);              /* Enable menu item           */
