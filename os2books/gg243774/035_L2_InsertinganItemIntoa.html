Inserting an Item Into a List Box

 hLBox = WinWindowFromID(hWnd,             /* Get list box window handle */
                         LB_LIST);

 ulIndex = WinInsertLboxItem(hLBox,        /* Insert list box item       */
                             LIT_END,      /* Insert at end of list      */
                             szItemText);  /* Item text                  */
